CONFIGURACION PERSONALIZADA
--------------------------------------
- GENERA UNA ARCHIVO PACKAGE JSON
yarn init -y 

- INSTALAMOS mysql2 
yarn add mysql2

- AGREGAMOS EN PACKAGE.JSON PERMITIRA UTILIZAR IMPORTS COMO EN REACT
"type": "module",

- INSTALAMOS cors ES UN MIDDLEWARE PERMITE HACER REQUEST A NUESTRO SERVER DESDE NUESTRO LOCALHOST OSEA DESDE REACT NATIVE
yarn add cors

- PERMITE CREAR VARIABLES DE ENTORNO PORQUE VAMOS A NESECITAR PASAR NUESTRO USERNAME Y PASSWORD A LA CONEXION MYSQL
yarn add dotenv

yarn add express

- INSTALAMOS COMO DEPENDENCIA DE DESARROLLO ESTO PERMITE CADA VEZ QUE HACEMOS UNA CAMBIO ESTE CORRERA AUTOMATICAMENTE NO TENEMOS QUE PARAR EL SERVIDOR Y VOLVERLO A CORRER
yarn add nodemon -D

- AGREGAMOS EN PACKAGE.JSON PARA ESPECIFICAR CUANDO CORRAMOS UN COMANDO DEV VAMOS A ESPECIFICAR QUE QUEREMOS HACER
"scripts": {"dev":"npx nodemon app.js"}

- CORREMOS EL COMANDO ANTERIOR PARA QUE CHEQUEE LOS ARCHIVOS COMO EL APP.JS
npm run dev
